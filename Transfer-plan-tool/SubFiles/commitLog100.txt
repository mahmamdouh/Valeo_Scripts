+/******************************************************************************
+ * \\brief   Name: uint32 SftyE2eTx_LimitPhyToRawVal(float32 value, const SftyE2eCanDataRngType sgnl)
+ * \
       Description:   This method creates appropriate unsigned integer value from float value
+ * \\param   [in] value:  The original physical value
+ * \\param   [in] sgnl: Signal Parameters.
+ * \\return  The uint32 raw value
+ *****************************************************************************/
+LOCAL_INLINE uint32 SftyE2eTx_LimitPhyToRawVal(float32 value, const SftyE2eCanDataRngType sgnl)
+{
+
+#ifdef SWIT_FC_Active
+if((SWIT_SftyE2eTx_EN_KEY==SWIT_First_HookEnableKey))
+   SWIT_SftyE2eTx_LimitPhyToRawVal_BgnHook ();
+#endif
+   /* ******** Variables declaration *********** */
+   uint32 retVal;
+   /* ************* Source code **************** */
+
+   retVal = MLIB_U32F32((value - sgnl.offs) * sgnl.invFact);
+
+   /* Limit value against upper threshold */
+   if(retVal > sgnl.maxVal)
+   {
+      retVal = sgnl.maxVal;
+   }
+   /* Limit value against lower threshold */
+   else if(retVal < sgnl.minVal)
+   {
+      retVal = sgnl.minVal;
+   }
+   else
+   {
+     /* Value is within limits -> do nothing */
+   }
+
+
+   /* Note: factor is expected to be != 0 */
+   return retVal;
+
+}
